                    case VK_RIGHT:
                    case VK_DOWN:
                    case VK_EXECUTE:
                    case VK_RETURN:
                    case VK_ESCAPE:

                        // IsDialogMessage should be called ProcessDialogMessage --

    // If it's one of our mnemonics, set focus to the appropriate hwnd
